import { describe, test } from "bun:test";
import { NeutralVisibilityHostFixture, VISIBILITY_LEASE_MILLISECONDS } from "./neutral-visibility-fixture";
const term = (key: string, idem: string) => ({
  key, idempotencyKey: idem,
  command: { executable: "/usr/bin/printf", arguments: ["x"], workingDirectory: "/tmp/n",
    completeEnvironment: [{ name: "TERM", value: "xterm-256color" }], descriptorMap: [] },
  terminalProfile: { inputMode: "literal" as const, echo: false, signalCharacters: true,
    softwareFlowControl: true, eofByte: 4, startByte: 17, stopByte: 19, hangupOnLastClose: true },
  initialWindow: { columns: 80, rows: 24, widthPixels: 800, heightPixels: 480 },
});
const s = { sessionId: "e3", process: { processId: 3, startToken: "3:a" } };
describe("edge3 orphaned lease", () => {
  test("expiry of a superseded generation's lease", async () => {
    const h = new NeutralVisibilityHostFixture();
    h.publishSnapshot({ source: s, inventoryRevision: "1", representedSessionKeys: ["k3"] });
    const a: any = await h.create({ terminal: term("k3", "i1"), visibility: { source: s, inventoryRevision: "1" } });
    const b: any = await h.create({ terminal: term("k3", "i2"), visibility: { source: s, inventoryRevision: "1" } });
    console.log("A:", a.result.session.incarnation, "B:", b.result.session.incarnation);
    console.log("leases held: A", h.currentLease(a.result.session) !== null, "| B", h.currentLease(b.result.session) !== null);
    try {
      await h.advance(VISIBILITY_LEASE_MILLISECONDS);
      console.log("advance() completed WITHOUT throwing");
      console.log("A expiry result:", JSON.stringify(h.expiryResult(a.result.session)?.state ?? null));
      console.log("B expiry result:", JSON.stringify(h.expiryResult(b.result.session)?.state ?? null));
    } catch (e: any) {
      console.log("advance() THREW:", e.constructor.name, e.message);
      console.log("A lease still held after throw:", h.currentLease(a.result.session) !== null);
      console.log("B lease still held after throw:", h.currentLease(b.result.session) !== null);
    }
  });
});
